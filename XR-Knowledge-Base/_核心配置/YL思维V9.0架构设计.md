class CoreActivationProtocol:
    """核心激活协议 - V9.0新增"""
    
    # 激活决策流程
    def activate_cores(question: str) -> List[Core]:
        """
        1. 问题解析 → 提取关键词与问题类型
        2. 核心匹配 → MatchCalculator计算匹配度
        3. 阈值判断 → 置信度门槛≥0.6
        4. 返回激活的核心列表

    # 区域联动规则
    def联动_regions(cores: List[Core]) -> Dict[Core, List[Region]]:
        """
        - 单核心问题：激活该核心下相关区域
        - 跨核心问题：按权重分配响应强度
        - 复合问题：多核心协同响应
        """
