## 三、V2.1三大增强机制（整合版）

### 3.1 层级折叠机制（含防护增强）

#### 3.1.1 折叠触发条件

| 层级 | 触发条件 | 折叠比例 | 保留内容 |
|:-----|:---------|:--------:|:---------|
| **L1-摘要** | 7天无激活 | 80% | 核心结论 + 索引指针 + 元数据 |
| **L2-索引** | 30天无激活 | 95% | 主题标签 + 检索关键词 + 存储位置 |
| **L3-元数据** | 180天无激活 | 99% | 基本标识 + 创建时间 + 关联关系 |
| **L4-引用** | 360天无激活 | 99.9% | 唯一ID + 外部引用计数 |

### 3.2 智能分割机制（含置信度增强）

#### 3.2.1 事件边界识别三信号

| 信号类型 | 检测内容 | 权重 | 示例 |
|:---------|:---------|:----:|:-----|
| **时间信号** | 时间戳跳跃、显式时间提及 | 0.2 | "昨天...今天..."、"上周...这周..." |
| **语义信号** | 话题转换词、语气变化 | 0.5 | "对了"、"换个话题"、"那件事..." |
| **实体信号** | 核心实体集合变化 | 0.3 | 新增/离开核心人物、场景切换 |

```

├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
```

```
│  ┌───────────────────────────────────────────────────────────────────────┐ │
│  │  场景                    │  前K个问题    │  V3.0稀疏激活              │ │

│                                                                              │
│  V3.0激活公式：                                                              │
│  ┌───────────────────────────────────────────────────────────────────────┐ │
│  │  激活条件：relevance_score ≥ threshold AND activated_count ≤ K_max   │ │
│  │                                                                       │ │
```

```python
def process_memory_input(user_input, context):
    """
    V3.0记忆输入完整流程（含精准度保障）
    """
    # ========== 阶段1：智能分割 ==========
    event_block = smart_segmentation(user_input, context)

    if event_block.boundary_confidence < 0.80:
        event_block.status = "PENDING_CONFIRMATION"
        event_block.flag_for_review()
    
    # ========== 阶段2：四层精准度校验 ==========
    validation_result = four_layer_validation(event_block)
    
    if validation_result.confidence < 0.95:
        return handle_low_confidence(validation_result)
    
    # ========== 阶段3：五维价值评估 ==========
    value_score = evaluate_memory_value(event_block, context)
    
    # 评估结果二次校验
    validation_result = four_layer_validation_for_score(value_score)
    
    # ========== 阶段4：层级路由 ==========
    target_layer = route_to_layer(value_score)
    
    # 迟滞机制防止抖动
    target_layer = apply_hysteresis(target_layer, value_score)
    
    # ========== 阶段5：折叠评估 ==========
    folding_decision = evaluate_folding(event_block, value_score)
    
    # 折叠防护检查
    if event_block.folding_protected:
        folding_decision = FOLDING_SKIP
    else:
        # 摘要预览确认
        preview = generate_summary_preview(event_block)
        if not confirm_with_user(preview):
            folding_decision = FOLDING_SKIP
    
    # ========== 阶段6：存储执行 ==========
    store_memory(event_block, target_layer, folding_decision)
    
    # ========== 阶段7：索引更新 ==========
    update_indexes(event_block, target_layer)
    
    return {
        "status": "SUCCESS",
        "event_block_id": event_block.event_id,
        "target_layer": target_layer,
        "folding_level": folding_decision.level,
        "confidence": validation_result.confidence
    }
```

### 6.2 稀疏激活算法

```python
def sparse_activation(question, context, config=None):
    """
    V3.0稀疏激活算法（含置信度门槛）
    """
    # 默认配置
    default_config = {
        "mode": "normal",  # strict/normal/loose/explore
        "threshold": 0.6,
        "k_max": 5,
        "default_neurons": ["谋士层", "主人层"]
    }
    
    config = {**default_config, **(config or {})}
    
    # ========== 步骤1：获取候选神经元 ==========
    all_neurons = get_all_neurons()
    
    # ========== 步骤2：计算相关性分数 ==========
    scored_neurons = []
    for neuron in all_neurons:
        relevance_score = calculate_relevance(question, neuron, context)
        
        # 置信度门槛判断
        if relevance_score >= config["threshold"]:
            scored_neurons.append({
                "neuron": neuron,
                "score": relevance_score
            })
    
    # ========== 步骤3：排序并截取 ==========
    scored_neurons.sort(key=lambda x: x["score"], reverse=True)
    
    # 稀疏激活：只取达到阈值的，不强行凑数
    activated = scored_neurons[:config["k_max"]]
    
    # ========== 步骤4：降级策略 ==========
    if len(activated) == 0:
        # 第一降级：降低阈值
        for lower_threshold in [0.5, 0.4]:
            for neuron in all_neurons:
                relevance_score = calculate_relevance(question, neuron, context)
                if relevance_score >= lower_threshold:
                    activated.append({"neuron": neuron, "score": relevance_score})
            
            if len(activated) > 0:
                break
        
        # 第二降级：激活默认神经元
        if len(activated) == 0:
            for default_neuron_name in config["default_neurons"]:
                default_neuron = get_neuron_by_name(default_neuron_name)
                activated.append({"neuron": default_neuron, "score": 0.5})
    
    # ========== 步骤5：四层校验 ==========
    validated_activation = []
    for item in activated:
        validation = four_layer_validation(item.neuron)
        if validation.confidence >= 0.95:
            validated_activation.append(item)
    
    return {
        "activated_neurons": validated_activation,
        "activation_count": len(validated_activation),
        "config_used": config,
        "degradation_used": len(activated) > config["k_max"]
    }
```

```python
def run_deduction_with_checkpoints(initial_state, deduction_steps):
    """
    V3.0推演检查点算法
    """
    checkpoints = []
    current_state = initial_state.copy()
    checkpoint_id = 0
    
    for step_idx, step in enumerate(deduction_steps):
        # ========== 检查点A：数据入口检查 ==========
        data_checkpoint = DataEntryCheckpoint(current_state, step.required_data)
        if not data_checkpoint.validate():
            return handle_data_validation_failure(data_checkpoint)
        
        # ========== 执行推演步骤 ==========
        new_state = step.execute(current_state)
        
        # ========== 检查点B：逻辑一致性检查 ==========
        logic_checkpoint = LogicConsistencyCheckpoint(
            previous_state=current_state,
            new_state=new_state,
            step=step
        )
        if not logic_checkpoint.validate():
            return handle_logic_inconsistency(logic_checkpoint)
        
        # ========== 检查点C：置信度评估 ==========
        confidence_checkpoint = ConfidenceCheckpoint(
            state=new_state,
            accumulated_confidence=calculate_accumulated_confidence(checkpoints)
        )
        confidence = confidence_checkpoint.validate()
        
        if confidence < 0.80:
            return handle_low_confidence(confidence_checkpoint)
        elif confidence < 0.95:
            trigger_multi_path_verification(new_state)
        
        # ========== 定期保存回滚点 ==========
        if step_idx % CHECKPOINT_INTERVAL == 0:
            rollback_checkpoint = RollbackCheckpoint(
                checkpoint_id=checkpoint_id,
                state=new_state.copy(),
                accumulated_confidence=confidence,
                assumptions=get_active_assumptions()
            )
            checkpoints.append(rollback_checkpoint)
            checkpoint_id += 1
    
    # ========== 最终检查点：累积置信度 ==========
    final_confidence = calculate_accumulated_confidence(checkpoints)
    
    return {
        "final_state": new_state,
        "confidence": final_confidence,
        "checkpoints": checkpoints,
        "uncertainty_tags": get_uncertainty_tags(new_state)
    }
```
